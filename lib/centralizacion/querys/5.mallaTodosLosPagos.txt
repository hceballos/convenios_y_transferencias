SELECT
	t.*,
	d.'Monto Cuota' as 'Deuda'
FROM		(

		SELECT 
			transferencias.* 
		FROM 
			transferencias
	)t

	LEFT JOIN (

		SELECT
			deuda.*
		FROM
			deuda
		WHERE
			deuda.'Tipo' = 'Debe'
			and deuda.'Estado Cuota' = 'Transferido'
			and deuda.'Estado Cuota' not NULL

	)d
	ON t.'cod_proyecto' = d.'cod_proyecto' 
	AND t.'numero_mes' = d.'numero_mes'
